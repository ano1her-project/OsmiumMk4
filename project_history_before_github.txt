        public void LoadFenPosition(string fenRecord)
        {
            string[] fenFields = fenRecord.Split(); // 6 fields
            // 1st field - Piece placement data
            string[] ranks = fenFields[0].Split('/');
            int y = 7;
            foreach (var rank in ranks)
            {
                int x = 0;
                char[] chars = rank.ToCharArray();
                foreach (var ch in chars)
                {                    
                    if (char.IsLetter(ch))
                    {
                        if (char.IsUpper(ch))
                        {
                            squares[x, y] = new SquareInfo(ch.ToString(), true);
                        }
                        else // Lowercase
                        {
                            squares[x, y] = new SquareInfo(ch.ToString().ToUpper(), false);
                        }
                        x++;
                    }
                    else // Number
                    {
                        for (int i = 0; i < Convert.ToInt32(ch.ToString()); i++)
                        {
                            squares[x, y] = new SquareInfo("x", false);
                            x++;
                        }
                    }
                }
                y--;
            }
            // 4th field - En passant target square
            if (fenFields[3] == "-")
            {
                enPassantTarget = Vector2.One * -1;
            }
            else
            {
                char[] chars = fenFields[3].ToCharArray();
                Dictionary<char, int> lettersToX = new Dictionary<char, int>
                {
                    { 'a', 0 }, { 'b', 1 }, { 'c', 2 }, { 'd', 3 }, { 'e', 4 }, { 'f', 5 }, { 'g', 6 }, { 'h', 7 }                        
                };
                enPassantTarget = new Vector2(Convert.ToInt32(lettersToX[chars[0]].ToString()), Convert.ToInt32(chars[1].ToString()) - 1);
            }
        }

        public RaycastHit PseudoRaycast(Vector2 start, Vector2 direction)
        {
            Vector2 inspectedSquarePos = start + direction;
            int i = 0;
            while (inspectedSquarePos.X >= 0 && inspectedSquarePos.X < 8 && inspectedSquarePos.Y >= 0 && inspectedSquarePos.Y < 8)
            {
                SquareInfo inspectedSquare = GetPieceAtSquare(inspectedSquarePos);
                if (inspectedSquare.piece != "x")
                {
                    return new RaycastHit(inspectedSquare, i);
                }
                i++;
                inspectedSquarePos += direction;
            }
            return new RaycastHit(new SquareInfo("x", true), i);
        }

        public SquareInfo GetPieceAtSquare(Vector2 inspectedSquarePos) // Also handles squares outside the bounds of the chessboard gracefully
        {
            if (inspectedSquarePos.X >= 0 && inspectedSquarePos.X < 8 && inspectedSquarePos.Y >= 0 && inspectedSquarePos.Y < 8)
            {
                return squares[Convert.ToInt32(inspectedSquarePos.X), Convert.ToInt32(inspectedSquarePos.Y)];
            }
            return new SquareInfo("OOB", false);
        }

        public bool IsInCheck(bool inspectingWhiteKing)
        {
            Vector2 kingPosition = GetPositionOfPiece("K", inspectingWhiteKing);
            SquareInfo enemyRook = new SquareInfo("R", !inspectingWhiteKing);
            SquareInfo enemyBishop = new SquareInfo("B", !inspectingWhiteKing);
            SquareInfo enemyQueen = new SquareInfo("Q", !inspectingWhiteKing);
            SquareInfo rayHit;
            // Orthogonal
            rayHit = PseudoRaycast(kingPosition, Vector2.UnitX).squareInfo;
            if (rayHit == enemyRook || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, Vector2.UnitY).squareInfo;
            if (rayHit == enemyRook || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, Vector2.UnitX * -1).squareInfo;
            if (rayHit == enemyRook || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, Vector2.UnitY * -1).squareInfo;
            if (rayHit == enemyRook || rayHit == enemyQueen)
                return true;
            // Diagonal
            rayHit = PseudoRaycast(kingPosition, Vector2.One).squareInfo;
            if (rayHit == enemyBishop || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, Vector2.One * -1).squareInfo;
            if (rayHit == enemyBishop || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, new Vector2(-1, 1)).squareInfo;
            if (rayHit == enemyBishop || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, new Vector2(1, -1)).squareInfo;
            if (rayHit == enemyBishop || rayHit == enemyQueen)
                return true;
            // Knights
            SquareInfo enemyKnight = new SquareInfo("N", !inspectingWhiteKing);
            if (GetPieceAtSquare(kingPosition + new Vector2(1, 2)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(2, 1)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(2, -1)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(1, -2)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(-1, -2)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(-2, -1)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(-2, 1)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(-1, 2)) == enemyKnight)
                return true;
            // Pawns
            SquareInfo enemyPawn = new SquareInfo("P", !inspectingWhiteKing);
            float relativePawnYPos = -1;
            if (!inspectingWhiteKing)
            {
                relativePawnYPos = 1;
            }
            if (GetPieceAtSquare(kingPosition + new Vector2(-1, relativePawnYPos)) == enemyPawn)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(1, relativePawnYPos)) == enemyPawn)
                return true;

            return false;
        }

        public Position ApplyMove(Move move) // Doesn't change the current position, a new position is returned.
        {
            Position newPosition = this.Clone(); // WE HAVE TO USE CLONE() so it's not just a fucken reference cuz that dosent work
            
            newPosition.squares[Convert.ToInt32(move.from.X), Convert.ToInt32(move.from.Y)] = new SquareInfo("x", false);

            return newPosition;
        }

        // Methods for evaluating a position
        public float EvaluatePosition()
        {
            // King safety
            float whiteKingSafety = Math.Abs(GetPositionOfPiece("K", true).X - 3.5f);
            float blackKingSafety = Math.Abs(GetPositionOfPiece("K", false).X - 3.5f);

            // Piece activity
            float whitePieceActivity = GetAllLegalMoves(true).Count;
            float blackPieceActivity = GetAllLegalMoves(false).Count;

            return GetMaterialBalance() * 1f + (whiteKingSafety - blackKingSafety) * 0.2f + (whitePieceActivity - blackPieceActivity) * 0.05f;
        }

        public int GetMaterialBalance()
        {
            int materialBalance = 0;

            Dictionary<string, int> pieceMaterialValues = new Dictionary<string, int>
            {
                { "R", 5 }, { "N", 3 }, { "B", 3 }, { "Q", 9 }, { "K", 0 }, { "P", 1 }, { "x", 0 }                        
            };

            foreach (var square in squares)
            {             
                if (square.isWhite)
                {
                    materialBalance += pieceMaterialValues[square.piece];
                }
                else
                {
                    materialBalance -= pieceMaterialValues[square.piece];
                }
            }

            return materialBalance;
        }

        public List<Move> GetAllMovesInGivenDirection(Vector2 start, Vector2 direction, bool whiteToMove)
        {
            List<Move> allMoves = new List<Move>();
            RaycastHit rayHit = PseudoRaycast(start, direction);
            for (int i = 0; i < rayHit.distance; i++)
            {
                allMoves.Add(new Move(start, start + direction * i, Vector2.One * -1, Vector2.One * -1));
            }
            if (rayHit.squareInfo.piece != "x" && rayHit.squareInfo.isWhite != whiteToMove)
            {
                allMoves.Add(new Move(start, start + direction * (rayHit.distance + 1), Vector2.One * -1, Vector2.One * -1));
            }
            return allMoves;
        }

        public List<Move> GetAllLegalMoves(bool whiteToMove)
        {            
            List<Move> allLegalMoves = new List<Move>();
            for (int x = 0; x < 8; x++)
            {
                for (int y = 0; y < 8; y++)
                {   
                    if (squares[x, y].isWhite != whiteToMove) // The piece isn't the same color as the player to move.
                    {
                        continue;
                    }     
                    switch (squares[x, y].piece)
                    {   
                        case "R":
                            allLegalMoves.AddRange(GetAllMovesInGivenDirection(new Vector2(x, y), Vector2.UnitX, whiteToMove));
                            allLegalMoves.AddRange(GetAllMovesInGivenDirection(new Vector2(x, y), Vector2.UnitY, whiteToMove));
                            allLegalMoves.AddRange(GetAllMovesInGivenDirection(new Vector2(x, y), Vector2.UnitX * -1, whiteToMove));
                            allLegalMoves.AddRange(GetAllMovesInGivenDirection(new Vector2(x, y), Vector2.UnitY * -1, whiteToMove));
                            break;

                        case "B":
                            allLegalMoves.AddRange(GetAllMovesInGivenDirection(new Vector2(x, y), Vector2.One, whiteToMove));
                            allLegalMoves.AddRange(GetAllMovesInGivenDirection(new Vector2(x, y), Vector2.One * -1, whiteToMove));
                            allLegalMoves.AddRange(GetAllMovesInGivenDirection(new Vector2(x, y), new Vector2(1, -1), whiteToMove));
                            allLegalMoves.AddRange(GetAllMovesInGivenDirection(new Vector2(x, y), new Vector2(-1, 1), whiteToMove));
                            break;

                        case "Q":
                            allLegalMoves.AddRange(GetAllMovesInGivenDirection(new Vector2(x, y), Vector2.UnitX, whiteToMove));
                            allLegalMoves.AddRange(GetAllMovesInGivenDirection(new Vector2(x, y), Vector2.UnitY, whiteToMove));
                            allLegalMoves.AddRange(GetAllMovesInGivenDirection(new Vector2(x, y), Vector2.UnitX * -1, whiteToMove));
                            allLegalMoves.AddRange(GetAllMovesInGivenDirection(new Vector2(x, y), Vector2.UnitY * -1, whiteToMove));
                            goto case "B";

                        case "N":
                            Vector2[] knightMoves = new Vector2[] { new Vector2(1, 2), new Vector2(2, 1), new Vector2(2, -1), new Vector2(1, -2), 
                                                                    new Vector2(-1, -2), new Vector2(-2, -1), new Vector2(-2, 1), new Vector2(-1, 2) };
                            for (int i = 0; i < 8; i++)
                            {
                                SquareInfo inspectedSquare = GetPieceAtSquare(new Vector2(x, y) + knightMoves[i]);
                                if (inspectedSquare.piece == "x" || (inspectedSquare.isWhite != whiteToMove && inspectedSquare.piece != "x" && inspectedSquare.piece != "OOB"))
                                {
                                    allLegalMoves.Add(new Move(new Vector2(x, y), new Vector2(x, y) + knightMoves[i], Vector2.One * -1, Vector2.One * -1));
                                }
                            }
                            break;

                        case "K":
                            Vector2[] kingMoves = new Vector2[] { Vector2.UnitY, Vector2.One, Vector2.UnitX, new Vector2(1, -1), 
                                                                  Vector2.UnitY * -1, Vector2.One * -1,  Vector2.UnitX * -1, new Vector2(-1, 1) };
                            for (int i = 0; i < 8; i++)
                            {
                                SquareInfo inspectedSquare = GetPieceAtSquare(new Vector2(x, y) + kingMoves[i]);
                                if (inspectedSquare.piece == "x" || (inspectedSquare.isWhite != whiteToMove && inspectedSquare.piece != "x" && inspectedSquare.piece != "OOB"))
                                {
                                    allLegalMoves.Add(new Move(new Vector2(x, y), new Vector2(x, y) + kingMoves[i], Vector2.One * -1, Vector2.One * -1));
                                }
                            }
                            break;

                        case "P":
                            float pawnMoveDirection = 1;
                            if (!whiteToMove)
                            {
                                pawnMoveDirection = -1;
                            }
                            Vector2 inspectedSquarePos = new Vector2(x, y) + Vector2.UnitY * pawnMoveDirection;
                            if (GetPieceAtSquare(inspectedSquarePos).piece == "x")
                            {
                                // One square move
                                allLegalMoves.Add(new Move(new Vector2(x, y), inspectedSquarePos, Vector2.One * -1, Vector2.One * -1));
                                inspectedSquarePos = new Vector2(x, y) + Vector2.UnitY * pawnMoveDirection * 2;
                                // Two squares move - En passant relevant!
                                if (y == 3.5 - pawnMoveDirection * 2.5 /*Is the pawn on its second rank?*/ && GetPieceAtSquare(new Vector2(x, y) + inspectedSquarePos).piece == "x")
                                {
                                    allLegalMoves.Add(new Move(new Vector2(x, y), inspectedSquarePos, Vector2.One * -1, new Vector2(x, y) + Vector2.UnitY * pawnMoveDirection));
                                }
                            }
                            for (int i = 1; i >= -1; i -= 2)
                            {
                                inspectedSquarePos = new Vector2(x, y) + new Vector2(i, pawnMoveDirection);
                                SquareInfo inspectedSquare = GetPieceAtSquare(inspectedSquarePos);
                                // Regular capturing
                                if (inspectedSquare.piece != "x" && inspectedSquare.piece != "OOB")
                                {
                                    allLegalMoves.Add(new Move(new Vector2(x, y), inspectedSquarePos, Vector2.One * -1, Vector2.One * -1));
                                } // En passant
                                else if (inspectedSquarePos == enPassantTarget)
                                {
                                    allLegalMoves.Add(new Move(new Vector2(x, y), inspectedSquarePos, inspectedSquarePos - Vector2.UnitY * pawnMoveDirection, Vector2.One * -1));
                                }
                            }
                            break;                          
                    }
                }
            }
            // Verify which moves won't leave the king in check
            for (int i = allLegalMoves.Count - 1; i >= 0 ; i--)
            {
                if (ApplyMove(allLegalMoves[i]).IsInCheck(whiteToMove))
                    allLegalMoves.RemoveAt(i);                        
            }
            return allLegalMoves;          
        }

    class MinimaxNode
    {
        public Position position;
        public float eval;
        public Move lastMovePlayed;
        public List<int> childIndexes;
        public bool whiteToMove;

        // Constructor
        public MinimaxNode(Position position, bool whiteToMove, Move lastMovePlayed)
        {
            this.position = position.Clone();
            this.whiteToMove = whiteToMove;
            this.lastMovePlayed = lastMovePlayed;
            this.childIndexes = new List<int>();
        }
    }

    class MinimaxTree
    {
        public MinimaxNode[][] tree;
        public int maxDepth;

        // Constructor
        public MinimaxTree(MinimaxNode rootNode, int maxDepth)
        {
            this.maxDepth = maxDepth;
            tree = new MinimaxNode[maxDepth + 1][];
            tree[0] = new MinimaxNode[1] { rootNode };
        }

        public void BuildTree() // Goes down the tree, building it.
        {
            for (int d = 0; d < maxDepth; d++)
            {
                List<MinimaxNode> thisTier = new List<MinimaxNode>();
                int i = 0;
                foreach (var node in tree[d])
                {
                    foreach (var move in node.position.GetAllLegalMoves(node.whiteToMove))
                    {
                        thisTier.Add(new MinimaxNode(node.position.ApplyMove(move), !node.whiteToMove, move));
                        node.childIndexes.Add(i);
                        i++;
                    }
                }
                tree[d + 1] = thisTier.ToArray();
            }
        }

        public Move FindBestMove() // Goes up the tree, evaluating each node, eventually reaching the node again.
        {
            for (int d = maxDepth; d >= 0 ; d--)
            {
                foreach (var node in tree[d])
                {
                    if (node.childIndexes.Count() == 0)
                    {
                        if (node.position.GetAllLegalMoves(node.whiteToMove).Count > 0)
                        {
                            node.eval = node.position.EvaluatePosition();
                        }
                        else // No legal moves
                        {
                            if (node.position.IsInCheck(node.whiteToMove)) // No legal moves and in check - Checkmate
                            {
                                node.eval = node.whiteToMove ? float.PositiveInfinity : float.NegativeInfinity;
                            }
                            else // No legal moves and is not in check - Stalemate
                            {
                                node.eval = 0;
                            }
                        }
                    }
                    else
                    {
                        List<MinimaxNode> childNodes = new List<MinimaxNode>();
                        foreach (var i in node.childIndexes)
                        {
                            childNodes.Add(tree[d + 1][i]);
                        }
                        node.eval = childNodes.OrderByDescending(n => n.eval).First().eval;
                    }
                }
            }
            // tree[1] are all children of the root node.
            return tree[1].OrderByDescending(n => n.eval).First().lastMovePlayed;
        }
    }

        public void LoadFenPosition(string fenRecord)
        {
            string[] fenFields = fenRecord.Split(); // 6 fields
            // 1st field - Piece placement data
            string[] ranks = fenFields[0].Split('/');
            int y = 7;
            foreach (var rank in ranks)
            {
                int x = 0;
                char[] chars = rank.ToCharArray();
                foreach (var ch in chars)
                {                    
                    if (char.IsLetter(ch))
                    {
                        if (char.IsUpper(ch))
                        {
                            squares[x, y] = new SquareInfo(ch.ToString(), true);
                        }
                        else // Lowercase
                        {
                            squares[x, y] = new SquareInfo(ch.ToString().ToUpper(), false);
                        }
                        x++;
                    }
                    else // Number
                    {
                        for (int i = 0; i < Convert.ToInt32(ch.ToString()); i++)
                        {
                            squares[x, y] = new SquareInfo("x", false);
                            x++;
                        }
                    }
                }
                y--;
            }
            // 4th field - En passant target square
            if (fenFields[3] == "-")
            {
                enPassantTarget = Vector2.One * -1;
            }
            else
            {
                char[] chars = fenFields[3].ToCharArray();
                Dictionary<char, int> lettersToX = new Dictionary<char, int>
                {
                    { 'a', 0 }, { 'b', 1 }, { 'c', 2 }, { 'd', 3 }, { 'e', 4 }, { 'f', 5 }, { 'g', 6 }, { 'h', 7 }                        
                };
                enPassantTarget = new Vector2(Convert.ToInt32(lettersToX[chars[0]].ToString()), Convert.ToInt32(chars[1].ToString()) - 1);
            }

        public RaycastHit PseudoRaycast(Vector2 start, Vector2 direction)
        {
            Vector2 inspectedSquarePos = start + direction;
            int i = 0;
            while (inspectedSquarePos.X >= 0 && inspectedSquarePos.X < 8 && inspectedSquarePos.Y >= 0 && inspectedSquarePos.Y < 8)
            {
                SquareInfo inspectedSquare = GetPieceAtSquare(inspectedSquarePos);
                if (inspectedSquare.piece != "x")
                {
                    return new RaycastHit(inspectedSquare, i);
                }
                i++;
                inspectedSquarePos += direction;
            }
            return new RaycastHit(new SquareInfo("x", true), i);
        }

        public SquareInfo GetPieceAtSquare(Vector2 inspectedSquarePos) // Also handles squares outside the bounds of the chessboard gracefully

                return squares[Convert.ToInt32(inspectedSquarePos.X), Convert.ToInt32(inspectedSquarePos.Y)];

            return new SquareInfo("OOB", false);

        public bool IsInCheck(bool inspectingWhiteKing)
        {
            Vector2 kingPosition = GetPositionOfPiece("K", inspectingWhiteKing);
            SquareInfo enemyRook = new SquareInfo("R", !inspectingWhiteKing);
            SquareInfo enemyBishop = new SquareInfo("B", !inspectingWhiteKing);
            SquareInfo enemyQueen = new SquareInfo("Q", !inspectingWhiteKing);
            SquareInfo rayHit;
            // Orthogonal
            rayHit = PseudoRaycast(kingPosition, Vector2.UnitX).squareInfo;
            if (rayHit == enemyRook || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, Vector2.UnitY).squareInfo;
            if (rayHit == enemyRook || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, Vector2.UnitX * -1).squareInfo;
            if (rayHit == enemyRook || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, Vector2.UnitY * -1).squareInfo;
            if (rayHit == enemyRook || rayHit == enemyQueen)
                return true;             
            // Diagonal
            rayHit = PseudoRaycast(kingPosition, Vector2.One).squareInfo;
            if (rayHit == enemyBishop || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, Vector2.One * -1).squareInfo;
            if (rayHit == enemyBishop || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, new Vector2(-1, 1)).squareInfo;
            if (rayHit == enemyBishop || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, new Vector2(1, -1)).squareInfo;
            if (rayHit == enemyBishop || rayHit == enemyQueen)
                return true;
            // Knights
            SquareInfo enemyKnight = new SquareInfo("N", !inspectingWhiteKing);
            if (GetPieceAtSquare(kingPosition + new Vector2(1, 2)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(2, 1)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(2, -1)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(1, -2)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(-1, -2)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(-2, -1)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(-2, 1)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(-1, 2)) == enemyKnight)
                return true;
            // Pawns
            SquareInfo enemyPawn = new SquareInfo("P", !inspectingWhiteKing);
            float relativePawnYPos = -1;
            if (!inspectingWhiteKing)
            {
                relativePawnYPos = 1;
            }
            if (GetPieceAtSquare(kingPosition + new Vector2(-1, relativePawnYPos)) == enemyPawn)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(1, relativePawnYPos)) == enemyPawn)
                return true;

            return false;
        }

        // Methods for evaluating a position
        public float EvaluatePosition()
        {
            // King safety
            float whiteKingSafety = Math.Abs(GetPositionOfPiece("K", true).X - 3.5f);
            float blackKingSafety = Math.Abs(GetPositionOfPiece("K", false).X - 3.5f);

            // Piece activity
            float whitePieceActivity = GetAllLegalMoves(true).Count;
            float blackPieceActivity = GetAllLegalMoves(false).Count;

            return GetMaterialBalance() * 1f + (whiteKingSafety - blackKingSafety) * 0.2f + (whitePieceActivity - blackPieceActivity) * 0.05f;
        }

        public int GetMaterialBalance()
        {
            int materialBalance = 0;

            Dictionary<string, int> pieceMaterialValues = new Dictionary<string, int>
            {
                { "R", 5 }, { "N", 3 }, { "B", 3 }, { "Q", 9 }, { "K", 0 }, { "P", 1 }, { "x", 0 }                        
            };

            foreach (var square in squares)
            {             
                if (square.isWhite)
                {
                    materialBalance += pieceMaterialValues[square.piece];
                }
                else
                {
                    materialBalance -= pieceMaterialValues[square.piece];
                }
            }

            return materialBalance;
        }

        public List<Move> GetAllMovesInGivenDirection(Vector2 start, Vector2 direction, bool whiteToMove)
        {
            List<Move> allMoves = new List<Move>();
            RaycastHit rayHit = PseudoRaycast(start, direction);
            for (int i = 1; i <= rayHit.distance; i++)
            {
                allMoves.Add(new Move(start, start + direction * i, Vector2.One * -1, Vector2.One * -1));
            }
            if (rayHit.squareInfo.piece != "x" && rayHit.squareInfo.isWhite != whiteToMove)
            {
                allMoves.Add(new Move(start, start + direction * (rayHit.distance + 1), Vector2.One * -1, Vector2.One * -1));
            }
            return allMoves;
        }

                                {

                                }
                            }
                            for (int i = 1; i >= -1; i -= 2)
                            {
                                inspectedSquarePos = new Vector2(x, y) + new Vector2(i, pawnMoveDirection);
                                SquareInfo inspectedSquare = GetPieceAtSquare(inspectedSquarePos);
                                // Regular capturing
                                if (inspectedSquare.piece != "x" && inspectedSquare.piece != "OOB")
                                {
                                    allLegalMoves.Add(new Move(new Vector2(x, y), inspectedSquarePos, Vector2.One * -1, Vector2.One * -1));
                                } // En passant
                                else if (inspectedSquarePos == enPassantTarget)
                                {
                                    allLegalMoves.Add(new Move(new Vector2(x, y), inspectedSquarePos, inspectedSquarePos - Vector2.UnitY * pawnMoveDirection, Vector2.One * -1));
                                }
                            }
                            break;                          
                    }
                }
            }
            // Verify which moves won't leave the king in check
            for (int i = allLegalMoves.Count - 1; i >= 0 ; i--)
            {
                if (ApplyMove(allLegalMoves[i]).IsInCheck(whiteToMove))
                    allLegalMoves.RemoveAt(i);                        
            }
            return allLegalMoves;          
        }

    class MinimaxNode
    {
        public Position position;
        public float eval;
        public Move lastMovePlayed;
        public List<int> childIndexes;
        public bool whiteToMove;

        // Constructor
        public MinimaxNode(Position position, bool whiteToMove, Move lastMovePlayed)
        {
            this.position = position.Clone();
            this.whiteToMove = whiteToMove;
            this.lastMovePlayed = lastMovePlayed;
            this.childIndexes = new List<int>();
        }
    }

    class MinimaxTree
    {
        public MinimaxNode[][] tree;
        public int maxDepth;

        // Constructor
        public MinimaxTree(MinimaxNode rootNode, int maxDepth)
        {
            this.maxDepth = maxDepth;
            tree = new MinimaxNode[maxDepth + 1][];
            tree[0] = new MinimaxNode[1] { rootNode };
        }

        public void BuildTree() // Goes down the tree, building it.
        {
            for (int d = 0; d < maxDepth; d++)
            {
                List<MinimaxNode> thisTier = new List<MinimaxNode>();
                int i = 0;
                foreach (var node in tree[d])
                {
                    foreach (var move in node.position.GetAllLegalMoves(node.whiteToMove))
                    {
                        thisTier.Add(new MinimaxNode(node.position.ApplyMove(move), !node.whiteToMove, move));
                        node.childIndexes.Add(i);
                        i++;
                    }
                }
                tree[d + 1] = thisTier.ToArray();
                Console.WriteLine($"Tier of depth {d + 1} built!");
            }
            Console.WriteLine("Tree built!");
        }

        public void BuildEval() // Goes up the tree, evaluating each node, eventually reaching the node again.
        {
            //
            for (int d = maxDepth; d >= 0 ; d--)
            {
                foreach (var node in tree[d])
                {
                    if (node.childIndexes.Count == 0)
                    {
                        if (node.position.GetAllLegalMoves(node.whiteToMove).Count > 0)
                        {
                            node.eval = node.position.EvaluatePosition();
                        }
                        else // No legal moves
                        {
                            if (node.position.IsInCheck(node.whiteToMove)) // No legal moves and in check - Checkmate
                            {
                                node.eval = node.whiteToMove ? -1000000f : 1000000f;
                            }
                            else // No legal moves and is not in check - Stalemate
                            {
                                node.eval = 0;
                            }
                        }
                    }
                    else
                    {
                        List<MinimaxNode> childNodes = new List<MinimaxNode>();
                        foreach (var i in node.childIndexes)
                        {
                            childNodes.Add(tree[d + 1][i]);
                        }                        
                        float proposedEval = childNodes.OrderByDescending(n => n.eval).First().eval;
                        if (Math.Abs(proposedEval) > 999000f) // Mate in 1000 or earlier
                        {
                            node.eval = proposedEval > 0 ? proposedEval - 0.5f : proposedEval + 0.5f;
                        }               
                    }
                }
                Console.WriteLine($"Tier of depth {d} evaluated!");
            }
        }

        public Move GetBestMove()
        {
            // tree[1] are all children of the root node.
            if (tree[1].Count() > 0)
            {
                return tree[1].OrderByDescending(n => n.eval).First().lastMovePlayed;
            }
            else
            {
                return null;
            }
        }
    }

        public void LoadFenPosition(string fenRecord)
        {
            string[] fenFields = fenRecord.Split(); // 6 fields
            // 1st field - Piece placement data
            string[] ranks = fenFields[0].Split('/');
            int y = 7;
            foreach (var rank in ranks)
            {
                int x = 0;
                char[] chars = rank.ToCharArray();
                foreach (var ch in chars)
                {                    
                    if (char.IsLetter(ch))
                    {
                        if (char.IsUpper(ch))
                        {
                            squares[x, y] = new SquareInfo(ch.ToString(), true);
                        }
                        else // Lowercase
                        {
                            squares[x, y] = new SquareInfo(ch.ToString().ToUpper(), false);
                        }
                        x++;
                    }
                    else // Number
                    {
                        for (int i = 0; i < Convert.ToInt32(ch.ToString()); i++)
                        {
                            squares[x, y] = new SquareInfo("x", false);
                            x++;
                        }
                    }
                }
                y--;
            }
            // 4th field - En passant target square
            if (fenFields[3] == "-")
            {
                enPassantTarget = Vector2.One * -1;
            }
            else
            {
                char[] chars = fenFields[3].ToCharArray();
                Dictionary<char, int> lettersToX = new Dictionary<char, int>
                {
                    { 'a', 0 }, { 'b', 1 }, { 'c', 2 }, { 'd', 3 }, { 'e', 4 }, { 'f', 5 }, { 'g', 6 }, { 'h', 7 }                        
                };
                enPassantTarget = new Vector2(Convert.ToInt32(lettersToX[chars[0]].ToString()), Convert.ToInt32(chars[1].ToString()) - 1);
            }
        }

        public RaycastHit PseudoRaycast(Vector2 start, Vector2 direction)
        {
            Vector2 inspectedSquarePos = start + direction;
            int i = 0;
            while (inspectedSquarePos.X >= 0 && inspectedSquarePos.X < 8 && inspectedSquarePos.Y >= 0 && inspectedSquarePos.Y < 8)
            {
                SquareInfo inspectedSquare = GetPieceAtSquare(inspectedSquarePos);
                if (inspectedSquare.piece != "x")
                {
                    return new RaycastHit(inspectedSquare, i);
                }
                i++;
                inspectedSquarePos += direction;
            }
            return new RaycastHit(new SquareInfo("x", true), i);
        }

        public SquareInfo GetPieceAtSquare(Vector2 inspectedSquarePos) // Also handles squares outside the bounds of the chessboard gracefully

            {
                return squares[Convert.ToInt32(inspectedSquarePos.X), Convert.ToInt32(inspectedSquarePos.Y)];
            }
            return new SquareInfo("OOB", false);
        }

        public bool IsInCheck(bool inspectingWhiteKing)
        {
            Vector2 kingPosition = GetPositionOfPiece("K", inspectingWhiteKing);
            SquareInfo enemyRook = new SquareInfo("R", !inspectingWhiteKing);
            SquareInfo enemyBishop = new SquareInfo("B", !inspectingWhiteKing);
            SquareInfo enemyQueen = new SquareInfo("Q", !inspectingWhiteKing);
            SquareInfo rayHit;
            // Orthogonal
            rayHit = PseudoRaycast(kingPosition, Vector2.UnitX).squareInfo;
            if (rayHit == enemyRook || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, Vector2.UnitY).squareInfo;
            if (rayHit == enemyRook || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, Vector2.UnitX * -1).squareInfo;
            if (rayHit == enemyRook || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, Vector2.UnitY * -1).squareInfo;
            if (rayHit == enemyRook || rayHit == enemyQueen)
                return true;             
            // Diagonal
            rayHit = PseudoRaycast(kingPosition, Vector2.One).squareInfo;
            if (rayHit == enemyBishop || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, Vector2.One * -1).squareInfo;
            if (rayHit == enemyBishop || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, new Vector2(-1, 1)).squareInfo;
            if (rayHit == enemyBishop || rayHit == enemyQueen)
                return true;
            rayHit = PseudoRaycast(kingPosition, new Vector2(1, -1)).squareInfo;
            if (rayHit == enemyBishop || rayHit == enemyQueen)
                return true;
            // Knights
            SquareInfo enemyKnight = new SquareInfo("N", !inspectingWhiteKing);
            if (GetPieceAtSquare(kingPosition + new Vector2(1, 2)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(2, 1)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(2, -1)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(1, -2)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(-1, -2)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(-2, -1)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(-2, 1)) == enemyKnight)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(-1, 2)) == enemyKnight)
                return true;
            // Pawns
            SquareInfo enemyPawn = new SquareInfo("P", !inspectingWhiteKing);
            float relativePawnYPos = -1;
            if (!inspectingWhiteKing)
            {
                relativePawnYPos = 1;
            }
            if (GetPieceAtSquare(kingPosition + new Vector2(-1, relativePawnYPos)) == enemyPawn)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(1, relativePawnYPos)) == enemyPawn)
                return true;

            return false;
        }

        // Methods for evaluating a position
        public float EvaluatePosition()
        {
            // King safety
            float whiteKingSafety = Math.Abs(GetPositionOfPiece("K", true).X - 3.5f);
            float blackKingSafety = Math.Abs(GetPositionOfPiece("K", false).X - 3.5f);

            // Piece activity
            float whitePieceActivity = GetAllLegalMoves(true).Count;
            float blackPieceActivity = GetAllLegalMoves(false).Count;

            return GetMaterialBalance() * 1f + (whiteKingSafety - blackKingSafety) * 0.2f + (whitePieceActivity - blackPieceActivity) * 0.05f;
        }

        public int GetMaterialBalance()
        {
            int materialBalance = 0;

            Dictionary<string, int> pieceMaterialValues = new Dictionary<string, int>
            {
                { "R", 5 }, { "N", 3 }, { "B", 3 }, { "Q", 9 }, { "K", 0 }, { "P", 1 }, { "x", 0 }                        
            };

            foreach (var square in squares)
            {             
                if (square.isWhite)
                {
                    materialBalance += pieceMaterialValues[square.piece];
                }
                else
                {
                    materialBalance -= pieceMaterialValues[square.piece];
                }
            }

            return materialBalance;
        }

        public List<Move> GetAllMovesInGivenDirection(Vector2 start, Vector2 direction, bool whiteToMove)
        {
            List<Move> allMoves = new List<Move>();
            RaycastHit rayHit = PseudoRaycast(start, direction);
            for (int i = 1; i <= rayHit.distance; i++)
            {
                allMoves.Add(new Move(start, start + direction * i, Vector2.One * -1, Vector2.One * -1));
            }
            if (rayHit.squareInfo.piece != "x" && rayHit.squareInfo.isWhite != whiteToMove)
            {
                allMoves.Add(new Move(start, start + direction * (rayHit.distance + 1), Vector2.One * -1, Vector2.One * -1));
            }
            return allMoves;
        }

                                // One square move
                                allLegalMoves.Add(new Move(new Vector2(x, y), inspectedSquarePos, Vector2.One * -1, Vector2.One * -1));
                                inspectedSquarePos = new Vector2(x, y) + Vector2.UnitY * pawnMoveDirection * 2;
                                // Two squares move - En passant relevant!
                                if (y == 3.5 - pawnMoveDirection * 2.5 /*Is the pawn on its second rank?*/ && GetPieceAtSquare(inspectedSquarePos).piece == "x")
                                {
                                    allLegalMoves.Add(new Move(new Vector2(x, y), inspectedSquarePos, Vector2.One * -1, new Vector2(x, y) + Vector2.UnitY * pawnMoveDirection));
                                }
                            }
                            for (int i = 1; i >= -1; i -= 2)
                            {
                                inspectedSquarePos = new Vector2(x, y) + new Vector2(i, pawnMoveDirection);
                                SquareInfo inspectedSquare = GetPieceAtSquare(inspectedSquarePos);
                                // Regular capturing
                                if (inspectedSquare.piece != "x" && inspectedSquare.piece != "OOB")
                                {
                                    allLegalMoves.Add(new Move(new Vector2(x, y), inspectedSquarePos, Vector2.One * -1, Vector2.One * -1));
                                } // En passant
                                else if (inspectedSquarePos == enPassantTarget)
                                {
                                    allLegalMoves.Add(new Move(new Vector2(x, y), inspectedSquarePos, inspectedSquarePos - Vector2.UnitY * pawnMoveDirection, Vector2.One * -1));
                                }
                            }
                            break;                          
                    }
                }
            }
            return allLegalMoves;   
        }

    class MinimaxNode
    {
        public Position position;
        public float eval;
        public Move lastMovePlayed;
        public List<int> childIndexes;
        public bool whiteToMove;

        // Constructor
        public MinimaxNode(Position position, bool whiteToMove, Move lastMovePlayed)
        {
            this.position = position.Clone();
            this.whiteToMove = whiteToMove;
            this.lastMovePlayed = lastMovePlayed;
            this.childIndexes = new List<int>();
        }
    }

    class MinimaxTree
    {
        public MinimaxNode[][] tree;
        public int maxDepth;

        // Constructor
        public MinimaxTree(MinimaxNode rootNode, int maxDepth)
        {
            this.maxDepth = maxDepth;
            tree = new MinimaxNode[maxDepth + 1][];
            tree[0] = new MinimaxNode[1] { rootNode };
        }

        public void BuildTree() // Goes down the tree, building it.
        {
            for (int d = 0; d < maxDepth; d++)
            {
                List<MinimaxNode> thisTier = new List<MinimaxNode>();
                int i = 0;
                foreach (var node in tree[d])
                {
                    foreach (var move in node.position.GetAllLegalMoves(node.whiteToMove))
                    {
                        thisTier.Add(new MinimaxNode(node.position.ApplyMove(move), !node.whiteToMove, move));
                        node.childIndexes.Add(i);
                        i++;
                    }
                }
                tree[d + 1] = thisTier.ToArray();
                Console.WriteLine($"Tier of depth {d + 1} built!");
            }
        }

        public void BuildEval() // Goes up the tree, evaluating each node, eventually reaching the node again.
        {
            //
            for (int d = maxDepth; d >= 0 ; d--)
            {
                foreach (var node in tree[d])
                {
                    if (node.childIndexes.Count == 0)
                    {
                        if (node.position.HasAnyLegalMoves(node.whiteToMove))
                        {
                            node.eval = node.position.EvaluatePosition();
                        }
                        else // No legal moves
                        {
                            if (node.position.IsInCheck(node.whiteToMove)) // No legal moves and in check - Checkmate
                            {
                                node.eval = node.whiteToMove ? -1000000f : 1000000f;
                            }
                            else // No legal moves and is not in check - Stalemate
                            {
                                node.eval = 0;
                            }
                        }
                    }
                    else
                    {
                        List<MinimaxNode> childNodes = new List<MinimaxNode>();
                        foreach (var i in node.childIndexes)
                        {
                            childNodes.Add(tree[d + 1][i]);
                        }                        
                        float proposedEval = childNodes.OrderByDescending(n => n.eval).First().eval;
                        if (Math.Abs(proposedEval) > 999000f) // Mate in 1000 or earlier
                        {
                            node.eval = proposedEval > 0 ? proposedEval - 0.5f : proposedEval + 0.5f;
                        }               
                    }
                }
                Console.WriteLine($"Tier of depth {d} evaluated!");
            }
        }

        public Move GetBestMove()
        {
            // tree[1] are all children of the root node.
            if (tree[1].Count() > 0)
            {
                return tree[1].OrderByDescending(n => n.eval).First().lastMovePlayed;
            }
            else
            {
                return null;
            }
        }
    }

        public RaycastHit PseudoRaycast(Vector2 start, Vector2 direction)
        {
            Vector2 inspectedSquarePos = start + direction;
            int i = 0;
            while (inspectedSquarePos.X >= 0 && inspectedSquarePos.X < 8 && inspectedSquarePos.Y >= 0 && inspectedSquarePos.Y < 8)

                SquareInfo inspectedSquare = GetPieceAtSquare(inspectedSquarePos);

                {

                }
                i++;
                inspectedSquarePos += direction;

            return new RaycastHit(new SquareInfo("x", true), i);
        }

        public bool IsInCheck(bool inspectingWhiteKing)
        {
            Vector2 kingPosition = GetPositionOfPiece("K", inspectingWhiteKing);
            SquareInfo enemyRook = new SquareInfo("R", !inspectingWhiteKing);
            SquareInfo enemyBishop = new SquareInfo("B", !inspectingWhiteKing);
            SquareInfo enemyQueen = new SquareInfo("Q", !inspectingWhiteKing);
            SquareInfo enemyKnight = new SquareInfo("N", !inspectingWhiteKing);
            SquareInfo rayHit;
            // Orthogonal
            for (int i = 0; i < 8; i += 2)
            {
                rayHit = PseudoRaycast(kingPosition, Consts.directions[i]).squareInfo;
                if (rayHit == enemyRook || rayHit == enemyQueen)
                return true;
            }     
            // Diagonal
            for (int i = 1; i < 8; i += 2)
            {
                rayHit = PseudoRaycast(kingPosition, Consts.directions[i]).squareInfo;
                if (rayHit == enemyBishop || rayHit == enemyQueen)
                return true;
            }  
            // Knights 
            for (int i = 0; i < 8; i++)
            {
                if (GetPieceAtSquare(kingPosition + Consts.knightDirections[i]) == enemyKnight)
                    return true;
            }          
            // Pawns
            SquareInfo enemyPawn = new SquareInfo("P", !inspectingWhiteKing);
            float relativePawnYPos = -1;
            if (!inspectingWhiteKing)
            {
                relativePawnYPos = 1;
            }
            if (GetPieceAtSquare(kingPosition + new Vector2(-1, relativePawnYPos)) == enemyPawn)
                return true;
            if (GetPieceAtSquare(kingPosition + new Vector2(1, relativePawnYPos)) == enemyPawn)
                return true;

            return false;
        }

        // Methods for evaluating a position
        public float EvaluatePosition()
        {
            // Material balance
            int materialBalance = 0;
            foreach (var square in squares)
            {             
                if (square.isWhite)
                {
                    materialBalance += Consts.pieceMaterialValues[square.piece];
                }
                else
                {
                    materialBalance -= Consts.pieceMaterialValues[square.piece];
                }
            }

            // King safety
            float whiteKingSafety = Math.Abs(GetPositionOfPiece("K", true).X - 3.5f);
            float blackKingSafety = Math.Abs(GetPositionOfPiece("K", false).X - 3.5f);

            // Piece activity
            float whitePieceActivity = GetAllLegalMoves(true).Count;
            float blackPieceActivity = GetAllLegalMoves(false).Count;

            return materialBalance * 1f + (whiteKingSafety - blackKingSafety) * 0.2f + (whitePieceActivity - blackPieceActivity) * 0.05f;
        }

        public List<Move> GetAllMovesInGivenDirection(Vector2 start, Vector2 direction, bool whiteToMove)
        {
            List<Move> allMoves = new List<Move>();
            RaycastHit rayHit = PseudoRaycast(start, direction);
            for (int i = 1; i <= rayHit.distance; i++)
            {
                allMoves.Add(new Move(start, start + direction * i, Vector2.One * -1, Vector2.One * -1));
            }
            if (rayHit.squareInfo.piece != "x" && rayHit.squareInfo.isWhite != whiteToMove)
            {
                allMoves.Add(new Move(start, start + direction * (rayHit.distance + 1), Vector2.One * -1, Vector2.One * -1));
            }
            return allMoves;
        }

        public List<Move> GetAllLegalMoves(bool whiteToMove) // Doesn't verify if the moves leave the king in check.

                    {
                        continue;
                    }     
                    switch (squares[x, y].piece)
                    {   
                        case "R":
                            for (int i = 0; i < 8; i += 2)
                            {
                                allLegalMoves.AddRange(GetAllMovesInGivenDirection(new Vector2(x, y), Consts.directions[i], whiteToMove));
                            }
                            break;

                        case "B":
                            for (int i = 1; i < 8; i += 2)
                            {
                                allLegalMoves.AddRange(GetAllMovesInGivenDirection(new Vector2(x, y), Consts.directions[i], whiteToMove));
                            }

    class MinimaxNode
    {
        public Position position;
        public float eval;
        public Move lastMovePlayed;
        public List<int> childIndexes;
        public bool whiteToMove;

        // Constructor
        public MinimaxNode(Position position, bool whiteToMove, Move lastMovePlayed)
        {
            this.position = position.Clone();
            this.whiteToMove = whiteToMove;
            this.lastMovePlayed = lastMovePlayed;
            this.childIndexes = new List<int>();
        }

        public MinimaxNode Clone()
        {
            MinimaxNode newNode = new(this.position, this.whiteToMove, this.lastMovePlayed);
            newNode.eval = this.eval;
            newNode.childIndexes = new List<int>(this.childIndexes);
            return newNode;
        }
    }

    class MinimaxTree
    {
        public MinimaxNode[][] tree;
        public int maxDepth;

        // Constructor
        public MinimaxTree(MinimaxNode rootNode, int maxDepth)
        {
            this.maxDepth = maxDepth;
            tree = new MinimaxNode[maxDepth + 1][];
            tree[0] = new MinimaxNode[1] { rootNode };
        }

        public void BuildTree(int startAtDepth) // Goes down the tree, building it.
        {
            for (int d = startAtDepth; d < maxDepth; d++)
            {
                List<MinimaxNode> thisTier = new List<MinimaxNode>();
                int i = 0;
                foreach (var node in tree[d])
                {
                    foreach (var move in node.position.GetAllLegalMoves(node.whiteToMove))
                    {
                        thisTier.Add(new MinimaxNode(node.position.ApplyMove(move), !node.whiteToMove, move));
                        node.childIndexes.Add(i);
                        i++;
                    }
                }
                tree[d + 1] = thisTier.ToArray();
                Console.WriteLine($"Tier of depth {d + 1} built!");
            }
        }

        public void BuildEval() // Goes up the tree, evaluating each node, eventually reaching the node again.
        {
            for (int d = maxDepth; d >= 0 ; d--)
            {
                foreach (var node in tree[d])
                {
                    if (node.childIndexes.Count == 0)
                    {
                        if (node.position.HasAnyLegalMoves(node.whiteToMove))
                        {
                            node.eval = node.position.EvaluatePosition();
                        }
                        else // No legal moves
                        {
                            if (node.position.IsInCheck(node.whiteToMove)) // No legal moves and in check - Checkmate
                            {
                                node.eval = node.whiteToMove ? -Consts.mateInZero : Consts.mateInZero;
                            }
                            else // No legal moves and is not in check - Stalemate
                            {
                                node.eval = 0;
                            }
                        }
                    }
                    else
                    {
                        List<MinimaxNode> childNodes = new List<MinimaxNode>();
                        foreach (var i in node.childIndexes)
                        {
                            childNodes.Add(tree[d + 1][i]);
                        }                        
                        float proposedEval = childNodes.OrderByDescending(n => n.eval).First().eval;
                        if (Math.Abs(proposedEval) > Consts.mateInOneThousand)
                        {
                            node.eval = proposedEval > 0 ? proposedEval - 0.5f : proposedEval + 0.5f;
                        }               
                    }
                }
                Console.WriteLine($"Tier of depth {d} evaluated!");
            }
        }

        public int FindNodeIndexByMove(int parentNodeDepth, int parentNodeIndex, Move move)
        {
            List<int> childIndexes = tree[parentNodeDepth][parentNodeIndex].childIndexes;
            foreach (var i in childIndexes)
            {
                if (tree[parentNodeDepth + 1][i].lastMovePlayed == move)
                    return i;
            }
            return -1;
        }

        public Vector2 FindNodePosByMoves(Move[] moves)
        {
            int d = 0;
            int i = 0;
            foreach (var move in moves)
            {
                i = FindNodeIndexByMove(d, i, move);
                d++;
            }
            return new(i, d);
        }

        public void Reroot(Vector2 newRootPos) // newRootPos.Y - depth, newRootPos.X - index
        {
            List<int> indexesToKeep = new List<int> { Convert.ToInt32(newRootPos.X) };
                        
            for (int d = Convert.ToInt32(newRootPos.Y); d <= maxDepth; d++)
            {
                // Trim all the unwanted nodes
                tree[d] = tree[d].Where((_, index) => indexesToKeep.Contains(index)).ToArray();
                // Get the list of indexes to keep from the next tier
                indexesToKeep = new();
                foreach (var node in tree[d])
                {
                    indexesToKeep.AddRange(node.childIndexes);
                }
            }
            // Move the tree accordingly           
            for (int dFrom = Convert.ToInt32(newRootPos.Y), dTo = 0; dTo <= maxDepth; dFrom++, dTo++)
            {
                // I can't just do tree[dTo] = tree[dFrom] because of reference type bullshit.
                tree[dTo] = Array.Empty<MinimaxNode>();
                if (dFrom > maxDepth)
                {
                    continue;
                }
                tree[dTo] = new MinimaxNode[tree[dFrom].Length];
                for (int i = 0; i < tree[dFrom].Length; i++)
                {
                    tree[dTo][i] = tree[dFrom][i].Clone();
                }
                
            }
        }

        public Move GetBestMove()
        {
            // tree[1] are all children of the root node.
            if (tree[1].Length > 0)
            {
                return tree[1].OrderByDescending(n => n.eval).First().lastMovePlayed;
            }
            else
            {
                return null;
            }
        }
    }

class Piece // Equivalent of the old SquareInfo
{
    public enum Type
    {
        Pawn,
        Knight,
        Bishop,
        Rook,
        Queen,
        King
    }

    public Type type;
    public bool isWhite;

    public Piece(Type type, bool isWhite)
    {
        this.type = type;
        this.isWhite = isWhite;
    }

    public override bool Equals(object? o)
    {
        if (o == null || o.GetType() != GetType())
            return false;
        return type == ((Piece)o).type && isWhite == ((Piece)o).isWhite;
    }

    public static bool operator ==(Piece? a, Piece? b)
    {
        return a.Equals(b);
    }

    public static bool operator !=(Piece? a, Piece? b)
    {
        return !(a == b);
    }

    public static Piece FromChar(char c)
    {
        return new(CharToType(c), char.IsUpper(c));
    }

    public char ToChar()
    {
        var c = TypeToChar();
        if (!isWhite)
            c = char.ToLower(c);
        return c;
    }

    override public string ToString()
    {
        return ToChar().ToString();
    }

    public static Type CharToType(char c)
    {
        switch (char.ToUpper(c))
        {
            case 'P':
                return Type.Pawn;
            case 'N':
                return Type.Knight;
            case 'B':
                return Type.Bishop;
            case 'R':
                return Type.Rook;
            case 'Q':
                return Type.Queen;
            default: // case 'K':
                return Type.King;
        }
    }

    public char TypeToChar()
    {
        switch (type)
        {
            case Type.Pawn:
                return 'P';
            case Type.Knight:
                return 'N';
            case Type.Bishop:
                return 'B';
            case Type.Rook:
                return 'R';
            case Type.Queen:
                return 'Q';
            default: // case Type.King:
                return 'K';
        }
    }

    public int GetValueFromType()
    {
        switch (type)
        {
            case Type.Pawn:
                return 1;
            case Type.Knight:
                return 3;
            case Type.Bishop:
                return 3;
            case Type.Rook:
                return 5;
            case Type.Queen:
                return 9;
            default: // case Type.King:
                return 0;
        }
    }

    public int GetValue()
    {
        return GetValueFromType() * (isWhite ? 1 : -1);
    }
}

        for (int y = 7; y >= 0; y--)
        {
            int x = 0;
            var chars = ranks[7 - y].ToCharArray(); // Here, 7 - y is used because it is the yth rank, but the 7-yth entry in the array of ranks. 
            for (int i = 0; x < 8; i++) // I *believe* x < 8 and i < chars.Length should have the same effect here.
            {
                if (char.IsLetter(chars[i]))
                {
                    output.board[x, y] = Piece.FromChar(chars[i]);
                    x++;
                    continue;
                }
                // If the program got here, chars[i] is a number.
                // In FEN, a number denotes some amount of empty spaces, so we simply skip that amount of x positions.
                x += chars[i];
            }
        }
        return output;

class MinimaxNode
{
    MinimaxNode? parent;
    readonly Position position;
    bool colorToMove;
    int depth;
    double? eval;

    public MinimaxNode(MinimaxNode parent, Position position, bool colorToMove, int depth)
    {
        this.position = position;
        this.colorToMove = colorToMove;
    }

    public double Evaluate(int maxDepth)
    {
        var moves = position.GetAllMoves();
        var evals = new double[moves.Length];
        for (int i = 0; i < moves.Length; i++)
        {
            var child = CreateChildNode(position.WithMove(moves[i]));
            evals[i] = child.depth >= maxDepth ? child.position.Evaluate() : child.Evaluate(maxDepth);
            // If we've hit max depth, get an evaluation for this position.
            // Otherwise, continue building the tree recursively.
        }
        return colorToMove ? evals.Max() : evals.Min();
    }

    MinimaxNode CreateChildNode(Position position)
    {
        return new(this, position, !colorToMove, depth + 1);
    }
}

class MinimaxNode
{
    MinimaxNode? parent;
    readonly Position position;
    bool colorToMove;
    int depth;
    double? eval;

    public MinimaxNode(MinimaxNode parent, Position position, bool colorToMove, int depth)
    {
        this.position = position;
        this.colorToMove = colorToMove;
    }

    public double Evaluate(int maxDepth)
    {
        var moves = position.GetAllLegalMovesFor(colorToMove);
        var evals = new double[moves.Length];
        for (int i = 0; i < moves.Length; i++)
        {
            var child = CreateChildNode(position.WithMove(moves[i]));
            evals[i] = child.depth >= maxDepth ? child.position.Evaluate() : child.Evaluate(maxDepth);
            // If we've hit max depth, get an evaluation for this position.
            // Otherwise, continue building the tree recursively.
        }
        return colorToMove ? evals.Max() : evals.Min();
    }

    MinimaxNode CreateChildNode(Position position)
    {
        return new(this, position, !colorToMove, depth + 1);
    }
}

using System.Numerics;

class Piece // Equivalent of the old SquareInfo
{
    public enum Type
    {
        Pawn,
        Knight,
        Bishop,
        Rook,
        Queen,
        King
    }

    public Type type;
    public bool isWhite;

    public Piece(Type type, bool isWhite)
    {
        this.type = type;
        this.isWhite = isWhite;
    }

    public static bool operator ==(Piece? a, Piece? b)
    {
        if (a is not null)
        {
            if (b is null)
                return false;
            return a.type == b.type && a.isWhite && b.isWhite;
        }
        // a is null
        return b is null;
    }

    public static bool operator !=(Piece? a, Piece? b)
    {
        return !(a == b);
    }

    public static Piece FromChar(char c)
    {
        return new(CharToType(c), char.IsUpper(c));
    }

    public char ToChar()
    {
        var c = TypeToChar();
        if (!isWhite)
            c = char.ToLower(c);
        return c;
    }

    override public string ToString()
    {
        return ToChar().ToString();
    }

    public static Type CharToType(char c)
    {
        switch (char.ToUpper(c))
        {
            case 'P':
                return Type.Pawn;
            case 'N':
                return Type.Knight;
            case 'B':
                return Type.Bishop;
            case 'R':
                return Type.Rook;
            case 'Q':
                return Type.Queen;
            default: // case 'K':
                return Type.King;
        }
    }

    public char TypeToChar()
    {
        switch (type)
        {
            case Type.Pawn:
                return 'P';
            case Type.Knight:
                return 'N';
            case Type.Bishop:
                return 'B';
            case Type.Rook:
                return 'R';
            case Type.Queen:
                return 'Q';
            default: // case Type.King:
                return 'K';
        }
    }

    public Vector[] GetMoveDirections()
    {
        switch (type)
        {
            case Type.Pawn:
                return Array.Empty<Vector>();
            case Type.Knight:
                return Vector.knightDirs;
            case Type.Bishop:
                return Vector.ordinals;
            case Type.Rook:
                return Vector.cardinals;
            case Type.Queen:
                return Vector.queenDirs;
            default: // case Type.King:
                return Vector.queenDirs;
        }
    }

    public int GetValueFromType()
    {
        switch (type)
        {
            case Type.Pawn:
                return 1;
            case Type.Knight:
                return 3;
            case Type.Bishop:
                return 3;
            case Type.Rook:
                return 5;
            case Type.Queen:
                return 9;
            default: // case Type.King:
                return 0;
        }
    }

    public int GetValue()
    {
        return GetValueFromType() * GetColorInt();
    }

    public int GetColorInt()
    {
        return isWhite ? 1 : -1;
    }

    public Piece DeepCopy()
    {
        return new(type, isWhite);
    }
}

    }

    {
        Vector[] outputFroms = new Vector[froms.Length];
        Vector[] outputTos = new Vector[tos.Length];

        else
            outputRemove = remove.DeepCopy();

        if (type == Type.King || type == Type.Knight || type == Type.Pawn)
            return 1;
        return 8;
    }

    public int GetValueFromType()
    {
        switch (type)
        {
            case Type.Pawn:
                return 1;
            case Type.Knight:
                return 3;
            case Type.Bishop:
                return 3;
            case Type.Rook:
                return 5;
            case Type.Queen:
                return 9;
            default: // case Type.King:
                return 0;
        }
    }

    public int GetValue()
    {
        return GetValueFromType() * GetColorInt();
    }

    public int GetColorInt()
    {
        return isWhite ? 1 : -1;
    }

    public Piece DeepCopy()
    {
        return new(type, isWhite);
    }
}

class Vector
{
    public int x; // file = column
    public int y; // rank = row

    public Vector(int x, int y)
    {
        this.x = x;
        this.y = y;
    }

    public static readonly Vector up = new(0, 1);
    public static readonly Vector right = new(1, 0);
    public static readonly Vector down = new(0, -1);
    public static readonly Vector left = new(-1, 0);
    public static readonly Vector zero = new(0, 0);
    public static readonly Vector one = new(1, 1);    

    public static readonly Vector[] cardinals = new Vector[] {up, right, down, left};
    public static readonly Vector[] ordinals = new Vector[] {one, new(1, -1), one * -1, new(-1, 1)};
    public static readonly Vector[] queenDirs = cardinals.Concat(ordinals).ToArray();
    public static readonly Vector[] knightDirs = new Vector[] {new(1, 2), new(2, 1), new(2, -1), new(1, -2), new(-1, -2), new(-2, -1), new(-2, 1), new(-1, 2)};

    public static Vector operator +(Vector a, Vector b)
    {
        return new(a.x + b.x, a.y + b.y);
    }

    public static Vector operator *(Vector a, int b)
    {
        return new(a.x * b, a.y * b);
    }

    public bool IsInBounds()
    {
        return x >= 0 && x < 8 && y >= 0 && y < 8;
    }

    public override string ToString()
    {
        return $"({x}, {y})";
    }

    public string ToSquare() 
    {
        return ((char)('a' + x)).ToString() + (y + 1);
    }

    public static Vector FromSquare(string s)
    {
        var chars = s.ToCharArray();
        return new(chars[0] - 'a', int.Parse(chars[1].ToString()) - 1);
    }

    public Vector2 ToVector2()
    {
        return new(x, y);
    }

    public Vector DeepCopy()
    {
        return new(x, y);
    }
}

class Move
// Note that this class doesn't include information about what the move did in the game
// - whether it was a capture, whether someone castled.
// Only the bare technical minimum, the variables that changed.
// As such, any "move" can be represented, such as pawn from a1 to h7, capturing e5 along the way.
{
    public Vector[] froms;
    public Vector[] tos;
    public Vector? remove;
    public Vector? newPassantTarget; 

    public Move(Vector[] froms, Vector[] tos, Vector remove, Vector newPassantTarget)
    {
        this.froms = froms;
        this.tos = tos;
        this.remove = remove;
        this.newPassantTarget = newPassantTarget;
    }
    public Move(Vector[] froms, Vector[] tos) : this(froms, tos, null, null)
    {}
    public Move(Vector from, Vector to, Vector remove, Vector newPassantTarget) : this(new Vector[] { from }, new Vector[] { to }, remove, newPassantTarget)
    {}
    public Move(Vector from, Vector to) : this(new Vector[] { from }, new Vector[] { to }, null, null)
    {}

    public Move DeepCopy()
    {
        Vector[] outputFroms = new Vector[froms.Length];
        Vector[] outputTos = new Vector[tos.Length];

        Vector? outputRemove;
        if (remove is null)
            outputRemove = null;
        else
            outputRemove = remove.DeepCopy();
        Vector? outputNewPassantTarget;
        if (newPassantTarget is null)
            outputNewPassantTarget = null;
        else
            outputNewPassantTarget = newPassantTarget.DeepCopy();

        for (int i = 0; i < froms.Length; i++)
        {
            outputFroms[i] = froms[i].DeepCopy();
            outputTos[i] = tos[i].DeepCopy();
        }
        return new(outputFroms, outputTos, outputRemove, outputNewPassantTarget);
    }
}

class Position
{
    readonly Piece?[,] board;
    Vector? passantTarget;

    static readonly public Position startingPosition = FromFEN("rnbqkbnr/pppppppp/8/8/8/8/PPPPPPPP/RNBQKBNR w KQkq - 0 1");

    public Position()
    {
        board = new Piece?[8, 8];
        passantTarget = null;
    }

    public static Position FromFEN(string fenRecord)
    {
        var output = new Position();
        string[] fields = fenRecord.Split();
        // 1st field - the board
        string[] ranks = fields[0].Split('/');
        for (int y = 7; y >= 0; y--)
        {
            int x = 0;
            var chars = ranks[7 - y].ToCharArray(); // Here, 7 - y is used because it is the yth rank, but the 7-yth entry in the array of ranks. 
            for (int i = 0; x < 8; i++) // I *believe* x < 8 and i < chars.Length should have the same effect here.
            {
                if (char.IsLetter(chars[i]))
                {
                    output.board[x, y] = Piece.FromChar(chars[i]);
                    x++;
                    continue;
                }
                // If the program got here, chars[i] is a number.
                // In FEN, a number denotes some amount of empty spaces, so we simply skip that amount of x positions.
                x += chars[i];
            }
        }
        return output;
    }

    public Piece? GetPieceAt(Vector pos)
    // This method differs from simply accessing board[x, y] in one crucial detail:
    // It checks whether the location makes sense - whether it's in bounds - in the first place.
    {
        if (!pos.IsInBounds())
            return null;
        return board[pos.x, pos.y];
    }

    public Vector? FindPiece(Piece piece)
    {
        for (int x = 0; x < 8; x++)
        {
            for (int y = 0; y < 8; y++)
            {
                if (board[x, y] is null) // Here, GetPieceAt() isn't necessary, it'll always be in bounds.
                    continue;
                if (board[x, y] == piece)
                    return new(x, y);
            }
        }
        return null;
    }

    public RaycastHit Raycast(Vector pos, Vector direction, int maxDistance)
    {
        RaycastHit output = new();
        while (true)
        {
            pos += direction;
            output.distance++;
            if (GetPieceAt(pos) is null && pos.IsInBounds() && output.distance < maxDistance)
                continue;
            output.piece = GetPieceAt(pos);
            return output;
        }
    }
    public RaycastHit Raycast(Vector pos, Vector direction)
    {
        return Raycast(pos, direction, 8);
    }

    public bool IsInCheck(bool color)
    {
        var kingPos = FindPiece(new(Piece.Type.King, color));
        if (kingPos is null)
            return false; // I guess if you don't have a king, you can't be in check.
        Piece queen = new(Piece.Type.Queen, !color);
        Piece otherThatApplies = new(Piece.Type.Rook, !color); // The first half of Vector.queenDirs is cardinals and the second is ordinals. So, at i = 4, we swap this for a bishop.
        for (int i = 0; i < 8; i++) // Here, 8 isn't the size of the board but the amount of queen directions.
        {
            var hit = Raycast(kingPos, Vector.queenDirs[i]);            
            if (i == 4)
                otherThatApplies.type = Piece.Type.Bishop;            
            if (hit.piece == queen || hit.piece == otherThatApplies)
                return true;
        }
        Piece knight = new(Piece.Type.Knight, !color);
        for (int i = 0; i < 8; i++) // Here, 8 isn't the size of the board but the amount of knight directions.
        {
            var hit = Raycast(kingPos, Vector.knightDirs[i], 1);
            if (hit.piece == knight)
                return true;
        }
        return false;
    }

    public List<Move> GetMovesOfPieceAt(Vector pos)
    {        
        var piece = GetPieceAt(pos);
        if (piece is null)
            return new();
        var output = new List<Move>();
        int maxRayDistance = piece.GetMaxMoveLength();
        foreach (var direction in piece.GetMoveDirections())
        {
            var hit = Raycast(pos, direction, maxRayDistance);
            if (hit.piece.isWhite != piece.isWhite)
                hit.distance++; // If the piece hit by the raycast is of the opposite color, going on that square is valid (as it is a capture).
            for (int i = 0; i < hit.distance; i++)
            {
                Move move = new(pos, pos + direction * i);
                if (!this.WithMove(move).IsInCheck(piece.isWhite))
                    output.Add(move);
            }
        }
        return output;
    }

    public int GetAmtOfMovesOfPieceAt(Vector pos)
    {        
        var piece = GetPieceAt(pos);
        if (piece is null)
            return 0;
        int output = 0;

        foreach (var direction in piece.GetMoveDirections())
        {
            var hit = Raycast(pos, direction, maxRayDistance);
            if (hit.piece.isWhite != piece.isWhite)

            }

    public List<Move> GetAllMovesForPlayer(bool color)
    {
        var output = new List<Move>();
        for (int x = 0; x < 8; x++)
        {
            for (int y = 0; y < 8; y++)
            {
                if (board[x, y] is null || board[x, y].isWhite != color)
                    continue;
                output.AddRange(GetMovesOfPieceAt(new(x, y)));
            }
        }
        return output;
    }

    public int GetAmtOfMovesForPlayer(bool color)
    {

        for (int x = 0; x < 8; x++)

    public bool GameEnds(bool colorToMove, out double? eval)
    {
        eval = null;
        if (GetAmtOfMovesForPlayer(colorToMove) != 0)
            return false;
        if (IsInCheck(colorToMove))
            eval = (double)Eval.Checkmate * (colorToMove ? -1 : 1);
        else
            eval = (double)Eval.Stalemate;
        return true;        
    }

    public Position WithMove(Move move) // Equivalent of the old ApplyMove()
    {
        Position output = DeepCopy();
        for (int i = 0; i < move.froms.Length; i++)
        {
            output.board[move.tos[i].x, move.tos[i].y] = output.board[move.froms[i].x, move.froms[i].y];
            output.board[move.froms[i].x, move.froms[i].y] = null;            
        }
        if (move.remove is not null)
            output.board[move.remove.x, move.remove.y] = null;
        if (move.newPassantTarget is not null)
            output.passantTarget = move.newPassantTarget;
        return output;
    }

class MinimaxNode
{
    MinimaxNode? parent;
    readonly Position position;
    bool colorToMove;
    int depth;
    double? eval;

    public MinimaxNode(MinimaxNode parent, Position position, bool colorToMove, int depth)
    {
        this.position = position;
        this.colorToMove = colorToMove;
    }

    // Polish this method later.
    public double Evaluate(int maxDepth)
    {
        if (position.GameEnds(colorToMove, out eval) && eval is not null)
            return (double)eval;    
        var moves = position.GetAllMovesForPlayer(colorToMove);
        var evals = new double[moves.Count];
        for (int i = 0; i < moves.Count; i++)
        {
            var child = CreateChildNode(position.WithMove(moves[i]));
            evals[i] = child.depth >= maxDepth ? child.position.Evaluate() : child.Evaluate(maxDepth);
            // If we've hit max depth, get an evaluation for this position.
            // Otherwise, continue building the tree recursively.
        }
        return colorToMove ? evals.Max() : evals.Min();
    }

    MinimaxNode CreateChildNode(Position position)
    {
        return new(this, position, !colorToMove, depth + 1);
    }
}
